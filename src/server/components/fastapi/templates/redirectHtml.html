Quick Start

Get up and running in seconds

server.ts
import { createServer } from 'xypriss';

// Initialize server with optimizations
const app = createServer({
  performance: {
    optimizationEnabled: true,
    preCompileRoutes: true
  },
  cache: {
    enabled: true,
    ttl: 3600
  },
  security: {
    rateLimit: true,
    cors: true
  }
});

// Define routes
app.get('/', (req, res) => {
  res.xJson({
    message: 'Hello from XyPrissJS!',
    version: '1.0.0',
    performance: '‚ö° Optimized'
  });
});

// Start server
app.start({{toPort}}, () => {
  console.log('üöÄ Server running on port {{toPort}}');
});
